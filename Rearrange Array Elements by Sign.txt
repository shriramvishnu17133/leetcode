class Solution:
    def rearrangeArray(self, nums: List[int]) -> List[int]:
        pos =[]
        neg=[]
        c=[]
        for i in nums:
            if i>0:
                pos.append (i)
            else :
                neg.append (i)
        for i in range (len(pos)) :
            c.append (pos[i])
            c.append (neg[i] )
        return c               
        return c
